005package org.anarres.qemu.qapi.common;
006
007import com.google.gson.annotations.SerializedName;
008
009/**
010 *
011 * @author shevek
012 */
013public class QApiResponse<V> extends QApiObject {
014
015    @SerializedName("return")
016    public V _return;
017    public String error;
018
019    public boolean isError() {
020        return error != null;
021    }
022
023    public V getResult() throws QApiException {
024        if (isError())
025            throw new QApiException(error);
026        return _return;
027    }
028}
