005package org.anarres.qemu.qapi.common;
006
007import com.fasterxml.jackson.databind.JsonNode;
008import com.fasterxml.jackson.databind.ObjectMapper;
009import java.io.BufferedReader;
010import java.io.Closeable;
011import java.io.EOFException;
012import java.io.IOException;
013import java.io.InputStreamReader;
014import java.io.OutputStreamWriter;
015import java.io.Writer;
016import java.net.InetAddress;
017import java.net.InetSocketAddress;
018import java.net.Socket;
019import java.nio.charset.Charset;
020import javax.annotation.Nonnegative;
021import javax.annotation.Nonnull;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025/**
026 * A connection to a QEmu process.
027 *
028 * @author shevek
029 */
030public class QApiConnection implements Closeable {
031
032    private static final Logger LOG = LoggerFactory.getLogger(QApiConnection.class);
033    private static final Charset ISO_8859_1 = Charset.forName("ISO-8859-1");
034    private final ObjectMapper mapper = new ObjectMapper();
035    private final Socket socket;
036    private final Writer output;
037    private final BufferedReader input;
038    private final QApiGreeting greeting;
039
040    public QApiConnection(@Nonnull Socket socket) throws IOException {
041        this.socket = socket;
042        this.output = new OutputStreamWriter(socket.getOutputStream(), ISO_8859_1);
043        this.input = new BufferedReader(new InputStreamReader(socket.getInputStream(), ISO_8859_1));
044        this.greeting = read(QApiGreeting.class);
045        /* QmpCapabilitiesCommand.Response capabilities = */ invoke(new QmpCapabilitiesCommand());
046        // assert !capabilities.isError();
047    }
048
049    public QApiConnection(@Nonnull InetAddress address, @Nonnegative int port) throws IOException {
050        this(new Socket(address, port));
051    }
052
053    public QApiConnection(@Nonnull String address, @Nonnegative int port) throws IOException {
054        this(InetAddress.getByName(address), port);
055    }
056
057    public QApiConnection(@Nonnull InetSocketAddress address) throws IOException {
058        this(address.getAddress(), address.getPort());
059    }
060
061    @Nonnull
062    public QApiGreeting getGreeting() {
063        return greeting;
064    }
065
066    @Nonnull
067    public QApiGreeting.QEmuVersion getQEmuVersion() {
068        return getGreeting().QMP.version.qemu;
069    }
070
071    @Nonnull
072    private <T> T read(@Nonnull Class<T> type) throws IOException {
073        for (;;) {
074            String line = input.readLine();
075            if (line == null)
076                throw new EOFException();
077            if (LOG.isDebugEnabled())
078                LOG.debug("<<<" + line);
079            JsonNode tree = mapper.readTree(line);
080            if (tree.get("event") != null)  // Could parse a QApiEvent.
081                continue;
082            return mapper.treeToValue(tree, type);
083        }
084        // return mapper.readValue(line, type);
085    }
086
087    @Nonnull
088    public <Response extends QApiResponse<?>> Response invoke(@Nonnull QApiCommand<?, Response> command) throws IOException {
089        String line = mapper.writeValueAsString(command);
090        if (LOG.isDebugEnabled())
091            LOG.debug(">>>" + line);
092        output.write(line);
093        output.write('\n');
094        output.flush();
095        Class<Response> returnType = command.getReturnType();
096        return read(returnType);
097    }
098
099    @Nonnull
100    public <Result, Response extends QApiResponse<Result>> Result call(@Nonnull QApiCommand<?, Response> command) throws IOException, QApiException {
101        return invoke(command).getResult();
102    }
103
104    @Override
105    public void close() throws IOException {
106        socket.close();
107    }
108}
