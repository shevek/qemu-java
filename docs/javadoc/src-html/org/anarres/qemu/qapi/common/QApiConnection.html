005package org.anarres.qemu.qapi.common;
006
007import com.fasterxml.jackson.databind.ObjectMapper;
008import java.io.BufferedReader;
009import java.io.Closeable;
010import java.io.IOException;
011import java.io.InputStreamReader;
012import java.io.OutputStreamWriter;
013import java.io.Writer;
014import java.net.InetSocketAddress;
015import java.net.Socket;
016import java.nio.charset.Charset;
017import javax.annotation.Nonnull;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021/**
022 * A connection to a QEmu process.
023 *
024 * @author shevek
025 */
026public class QApiConnection implements Closeable {
027
028    private static final Logger LOG = LoggerFactory.getLogger(QApiConnection.class);
029    private static final Charset ISO_8859_1 = Charset.forName("ISO-8859-1");
030    private final ObjectMapper mapper = new ObjectMapper();
031    private final Socket socket;
032    private final Writer output;
033    private final BufferedReader input;
034    private final QApiGreeting greeting;
035
036    public QApiConnection(@Nonnull Socket socket) throws IOException {
037        this.socket = socket;
038        this.output = new OutputStreamWriter(socket.getOutputStream(), ISO_8859_1);
039        this.input = new BufferedReader(new InputStreamReader(socket.getInputStream(), ISO_8859_1));
040        this.greeting = read(QApiGreeting.class);
041        /* QmpCapabilitiesCommand.Response capabilities = */ invoke(new QmpCapabilitiesCommand());
042        // assert !capabilities.isError();
043    }
044
045    public QApiConnection(@Nonnull InetSocketAddress address) throws IOException {
046        this(new Socket(address.getAddress(), address.getPort()));
047    }
048
049    @Nonnull
050    public QApiGreeting getGreeting() {
051        return greeting;
052    }
053
054    @Nonnull
055    public QApiGreeting.QEmuVersion getQEmuVersion() {
056        return getGreeting().QMP.version.qemu;
057    }
058
059    @Nonnull
060    private <T> T read(@Nonnull Class<T> type) throws IOException {
061        String line = input.readLine();
062        if (LOG.isDebugEnabled())
063            LOG.debug("<<<" + line);
064        return mapper.readValue(line, type);
065    }
066
067    @Nonnull
068    public <Response extends QApiResponse<?>> Response invoke(@Nonnull QApiCommand<?, Response> command) throws IOException {
069        String line = mapper.writeValueAsString(command);
070        if (LOG.isDebugEnabled())
071            LOG.debug(">>>" + line);
072        output.write(line);
073        output.write('\n');
074        output.flush();
075        Class<Response> returnType = command.getReturnType();
076        return read(returnType);
077    }
078
079    @Nonnull
080    public <Result, Response extends QApiResponse<Result>> Result call(@Nonnull QApiCommand<?, Response> command) throws IOException, QApiException {
081        return invoke(command).getResult();
082    }
083
084    @Override
085    public void close() throws IOException {
086        socket.close();
087    }
088}
