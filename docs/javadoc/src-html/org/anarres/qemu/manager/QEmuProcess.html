005package org.anarres.qemu.manager;
006
007import com.google.common.base.MoreObjects;
008import java.io.IOException;
009import java.net.ConnectException;
010import java.net.InetSocketAddress;
011import java.net.NoRouteToHostException;
012import java.net.UnknownServiceException;
013import java.util.concurrent.TimeUnit;
014import javax.annotation.CheckForNull;
015import javax.annotation.Nonnull;
016import org.anarres.qemu.qapi.api.QuitCommand;
017import org.anarres.qemu.qapi.common.QApiConnection;
018import org.anarres.qemu.qapi.common.QApiException;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022/**
023 *
024 * @author shevek
025 */
026public class QEmuProcess {
027
028    private static final Logger LOG = LoggerFactory.getLogger(QEmuProcess.class);
029    private final Process process;
030    private final InetSocketAddress monitor;
031    private QApiConnection connection;
032    private final Object lock = new Object();
033    // private final IOBuffer stdout = new IOBuffer();
034    // private final IOBuffer stderr = new IOBuffer();
035
036    public QEmuProcess(@Nonnull Process process, @CheckForNull InetSocketAddress monitor) {
037        this.process = process;
038        this.monitor = monitor;
039
040        // new IOThread(process.getInputStream(), stdout).start();
041        // new IOThread(process.getErrorStream(), stderr).start();
042    }
043
044    /**
045     * @throws NoRouteToHostException if the process is terminated.
046     * @throws UnknownServiceException if the process has no known monitor address.
047     */
048    @Nonnull
049    public QApiConnection getConnection() throws IOException {
050        if (monitor == null)
051            throw new UnknownServiceException("No monitor address known.");
052
053        try {
054            int exitValue = process.exitValue();
055            throw new NoRouteToHostException("Process terminated with exit code " + exitValue);
056        } catch (IllegalThreadStateException e) {
057        }
058
059        synchronized (lock) {
060            if (connection != null)
061                return connection;
062            connection = new QApiConnection(monitor);
063            return connection;
064        }
065    }
066
067    @Nonnull
068    public QApiConnection getConnection(long timeout, @Nonnull TimeUnit unit) throws IOException, InterruptedException {
069        long end = System.currentTimeMillis() + unit.toMillis(timeout);
070        while (end > System.currentTimeMillis()) {
071            try {
072                return getConnection();
073            } catch (ConnectException e) {
074                LOG.warn("Failed to connect to " + this + ": " + e);
075            }
076
077            long delay = Math.min(500, end - System.currentTimeMillis());
078            Thread.sleep(delay);
079        }
080
081        return getConnection(); // This last call blows our timing out of the window. :-(
082    }
083
084    public void destroy() throws IOException, QApiException {
085        try {
086            QApiConnection c = getConnection();
087            if (c != null) {
088                c.call(new QuitCommand());
089                c.close();
090            }
091        } catch (IllegalStateException e) {
092            LOG.warn("Cannot destroy " + this, e);
093        } catch (IOException e) {
094            LOG.warn("Cannot destroy " + this, e);
095        }
096        process.destroy();
097    }
098
099    @Override
100    public String toString() {
101        return MoreObjects.toStringHelper(this)
102                .add("process", process)
103                // .add("stdout", stdout)
104                // .add("stderr", stderr)
105                .toString();
106    }
107}
