005package org.anarres.qemu.exec;
006
007import java.io.File;
008import java.util.Arrays;
009import java.util.List;
010import javax.annotation.Nonnegative;
011import javax.annotation.Nonnull;
012
013/**
014 *
015 * @author shevek
016 */
017public class QEmuDriveOption extends AbstractQEmuOption {
018    // file=file,index=0,media=disk
019
020    public static enum Interface {
021
022        ide, scsi, sd, mtd, floppy, pflash, virtio
023    }
024
025    public static enum Media {
026
027        disk, cdrom
028    }
029
030    public static enum Cache {
031
032        none, writeback, unsafe, directsync, writethrough
033    }
034
035    public static enum Aio {
036
037        threads("threads"),
038        _native("native");
039        private final String text;
040        /* pp */ Aio(String text) {
041            this.text = text;
042        }
043
044        @Override
045        public String toString() {
046            return text;
047        }
048    }
049
050    public static enum Discard {
051
052        off, on
053    }
054
055    public static enum ErrorAction {
056
057        ignore, stop, report, enospc
058    }
059
060    public static enum CopyOnRead {
061
062        off, on
063    }
064    public File file;
065    // bus, unit
066    public int index;
067    public Interface iface;
068    public Media media;
069    public Cache cache;
070    public Aio aio;
071    public Discard discard;
072    public ErrorAction werror;
073    public ErrorAction rerror;
074    public boolean readonly;
075    public CopyOnRead copyOnRead;
076
077    public QEmuDriveOption() {
078    }
079
080    public QEmuDriveOption(int index, File file) {
081        this.file = file;
082        this.index = index;
083    }
084
085    public QEmuDriveOption(int index, String file) {
086        this(index, new File(file));
087    }
088
089    @Nonnull
090    public QEmuDriveOption withFile(@Nonnull File file) {
091        this.file = file;
092        return this;
093    }
094
095    @Nonnull
096    public QEmuDriveOption withIndex(@Nonnegative int index) {
097        this.index = index;
098        return this;
099    }
100
101    @Nonnull
102    public QEmuDriveOption withInterface(@Nonnegative Interface iface) {
103        this.iface = iface;
104        return this;
105    }
106
107    @Override
108    public void appendTo(List<? super String> line) {
109        StringBuilder buf = new StringBuilder();
110        appendTo(buf, "file", file);
111        appendTo(buf, "index", index);
112        appendTo(buf, "if", iface);
113        appendTo(buf, "media", media);
114        appendTo(buf, "cache", cache);
115        appendTo(buf, "aio", aio);
116        appendTo(buf, "discard", discard);
117        appendTo(buf, "werror", werror);
118        appendTo(buf, "rerror", rerror);
119        if (readonly)
120            buf.append("readonly");
121        appendTo(buf, "copy-on-read", copyOnRead);
122
123        line.addAll(Arrays.asList("-drive", buf.toString()));
124    }
125}
