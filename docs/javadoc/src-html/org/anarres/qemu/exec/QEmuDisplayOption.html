005package org.anarres.qemu.exec;
006
007import java.util.List;
008import javax.annotation.Nonnull;
009
010/**
011 *
012 * @author shevek
013 */
014public class QEmuDisplayOption extends AbstractQEmuOption {
015
016    public static enum DisplayType {
017
018        sdl, curses, none, gtk, vnc, nographic
019    }
020
021    public static enum VgaType {
022
023        cirrus, std, vmware, qxl, tcx, cg3, none;
024    }
025    private final DisplayType displayType;
026    private VgaType vgaType;
027
028    public QEmuDisplayOption(@Nonnull DisplayType displayType) {
029        this.displayType = displayType;
030    }
031
032    @Nonnull
033    public QEmuDisplayOption withVgaType(@Nonnull VgaType vgaType) {
034        this.vgaType = vgaType;
035        return this;
036    }
037
038    @Override
039    public void appendTo(List<? super String> line) {
040        if (displayType == DisplayType.nographic)
041            add(line, "-nographic");
042        else
043            add(line, "-display", displayType);
044        if (vgaType != null)
045            add(line, "-vga", vgaType);
046    }
047}
