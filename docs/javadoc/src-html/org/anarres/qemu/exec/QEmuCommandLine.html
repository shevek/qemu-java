005package org.anarres.qemu.exec;
006
007import java.util.ArrayList;
008import java.util.Arrays;
009import java.util.List;
010import javax.annotation.CheckForNull;
011import javax.annotation.Nonnull;
012
013/**
014 *
015 * @author shevek
016 */
017public class QEmuCommandLine {
018
019    private QEmuArchitecture architecture;
020    private final List<QEmuOption> options = new ArrayList<QEmuOption>();
021
022    public QEmuCommandLine(@Nonnull QEmuArchitecture architecture) {
023        this.architecture = architecture;
024    }
025
026    @Nonnull
027    public QEmuArchitecture getArchitecture() {
028        return architecture;
029    }
030
031    public void setArchitecture(@Nonnull QEmuArchitecture architecture) {
032        this.architecture = architecture;
033    }
034
035    @Nonnull
036    public List<? extends QEmuOption> getOptions() {
037        return options;
038    }
039
040    @Nonnull
041    public <T extends QEmuOption> List<? extends T> getOptions(@Nonnull Class<T> type) {
042        List<T> out = new ArrayList<T>();
043        for (QEmuOption option : getOptions())
044            if (type.isInstance(option))
045                out.add(type.cast(option));
046        return out;
047    }
048
049    @CheckForNull
050    public <T extends QEmuOption> T getOption(@Nonnull Class<T> type) {
051        for (QEmuOption option : getOptions())
052            if (type.isInstance(option))
053                return type.cast(option);
054        return null;
055    }
056
057    public void addOptions(@Nonnull Iterable<? extends QEmuOption> options) {
058        for (QEmuOption option : options)
059            this.options.add(option);
060    }
061
062    public void addOptions(@Nonnull QEmuOption... options) {
063        addOptions(Arrays.asList(options));
064    }
065
066    @Nonnull
067    public List<String> toCommandWords() {
068        List<String> line = new ArrayList<String>();
069        line.add(architecture.getCommand());
070        for (QEmuOption option : options)
071            option.appendTo(line);
072        return line;
073    }
074
075    @Override
076    public String toString() {
077        StringBuilder buf = new StringBuilder();
078        for (String word : toCommandWords()) {
079            if (buf.length() > 0)
080                buf.append(' ');
081            buf.append(word);
082        }
083        return buf.toString();
084    }
085}
