005package org.anarres.qemu.exec;
006
007import java.io.IOException;
008import java.util.ArrayList;
009import java.util.Arrays;
010import java.util.List;
011import javax.annotation.CheckForNull;
012import javax.annotation.Nonnull;
013import org.anarres.qemu.exec.recipe.QEmuRecipe;
014import org.anarres.qemu.exec.util.QEmuCommandLineUtils;
015import org.anarres.qemu.exec.util.QEmuIdAllocator;
016import org.anarres.qemu.exec.util.QEmuOptionsList;
017
018/**
019 * An execution command line for a QEmu process.
020 *
021 * A QEmuCommandLine is a tree of {@link QEmuOption}, not a list
022 * (see {@link QEmuOptionsList}).
023 *
024 * Disk and bus IDs within a QEmuOption may be allocated using the embedded
025 * {@link QEmuIdAllocator} in order to guarantee uniqueness.
026 *
027 * For convenience, composites of commonly used options are provided as
028 * {@link QEmuRecipe}s.
029 *
030 * @author shevek
031 */
032public class QEmuCommandLine {
033
034    private final QEmuIdAllocator allocator;
035    private QEmuArchitecture architecture;
036    private final List<QEmuOption> options = new ArrayList<QEmuOption>();
037
038    private QEmuCommandLine(@Nonnull QEmuArchitecture architecture, @Nonnull QEmuIdAllocator allocator) {
039        this.allocator = allocator;
040        this.architecture = architecture;
041    }
042
043    public QEmuCommandLine(@Nonnull QEmuArchitecture architecture) {
044        this(architecture, new QEmuIdAllocator());
045    }
046
047    public QEmuCommandLine(@Nonnull QEmuCommandLine prototype) {
048        this(prototype.getArchitecture(), new QEmuIdAllocator(prototype.getAllocator()));
049        addOptions(prototype.getOptions());
050    }
051
052    @Nonnull
053    public QEmuArchitecture getArchitecture() {
054        return architecture;
055    }
056
057    public void setArchitecture(@Nonnull QEmuArchitecture architecture) {
058        this.architecture = architecture;
059    }
060
061    @Nonnull
062    public QEmuIdAllocator getAllocator() {
063        return allocator;
064    }
065
066    /** Returns all the options in this command line. */
067    @Nonnull
068    public List<? extends QEmuOption> getOptions() {
069        return options;
070    }
071
072    private static <T extends QEmuOption> void getOptions(@Nonnull List<? super T> out, @Nonnull Iterable<? extends QEmuOption> in, @Nonnull Class<T> type) {
073        for (QEmuOption option : in) {
074            if (type.isInstance(option))
075                out.add(type.cast(option));
076            if (option instanceof QEmuOptionsList)
077                getOptions(out, (QEmuOptionsList) option, type);
078        }
079    }
080
081    /**
082     * Returns all options of the specified type in this command line.
083     */
084    @Nonnull
085    public <T extends QEmuOption> List<? extends T> getOptions(@Nonnull Class<T> type) {
086        List<T> out = new ArrayList<T>();
087        getOptions(out, getOptions(), type);
088        return out;
089    }
090
091    /**
092     * Returns the first option of the specified type in this command line.
093     *
094     * If there are no options of type <code>type</code>, null is returned.
095     * If there is more than one option of type <code>type</code>, the subsequent
096     * options are ignored.
097     */
098    @CheckForNull
099    public <T extends QEmuOption> T getOption(@Nonnull Class<T> type) {
100        List<? extends T> options = getOptions(type);
101        if (options.isEmpty())
102            return null;
103        return options.iterator().next();
104    }
105
106    public void addOptions(@Nonnull Iterable<? extends QEmuOption> options) {
107        for (QEmuOption option : options)
108            if (option != null)
109                this.options.add(option);
110    }
111
112    public void addOptions(@Nonnull QEmuOption... options) {
113        addOptions(Arrays.asList(options));
114    }
115
116    @Nonnull
117    public List<String> toCommandWords() {
118        List<String> line = new ArrayList<String>();
119        line.add(architecture.getCommand());
120        for (QEmuOption option : options)
121            option.appendTo(line);
122        return line;
123    }
124
125    /**
126     * Executes this QEmuCommandLine and returns a new {@link Process}.
127     */
128    @Nonnull
129    public Process exec() throws IOException {
130        List<String> commandWords = toCommandWords();
131        ProcessBuilder builder = new ProcessBuilder(commandWords);
132        // TODO: Use Redirect to send the I/O somewhere useful.
133        QEmuCommandLineUtils.redirectIO(builder);
134        return builder.start();
135    }
136
137    @Override
138    public String toString() {
139        StringBuilder buf = new StringBuilder();
140        for (String word : toCommandWords()) {
141            if (buf.length() > 0)
142                buf.append(' ');
143            buf.append(word);
144        }
145        return buf.toString();
146    }
147}
