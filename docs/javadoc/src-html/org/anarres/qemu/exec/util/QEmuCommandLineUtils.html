005package org.anarres.qemu.exec.util;
006
007import java.net.InetSocketAddress;
008import java.util.UUID;
009import javax.annotation.CheckForNull;
010import javax.annotation.Nonnull;
011import org.anarres.qemu.exec.QEmuChardevOption;
012import org.anarres.qemu.exec.QEmuCommandLine;
013import org.anarres.qemu.exec.QEmuIdOption;
014import org.anarres.qemu.exec.QEmuMonitorOption;
015import org.anarres.qemu.exec.host.chardev.TcpCharDevice;
016
017/**
018 *
019 * @author shevek
020 */
021public class QEmuCommandLineUtils {
022
023    @CheckForNull
024    public static UUID getUuid(@Nonnull QEmuCommandLine commandLine) {
025        QEmuIdOption idOption = commandLine.getOption(QEmuIdOption.class);
026        if (idOption == null)
027            return null;
028        return idOption.getUuid();
029    }
030
031    @CheckForNull
032    public static InetSocketAddress getMonitorAddress(@Nonnull QEmuCommandLine commandLine) {
033        QEmuMonitorOption monitorOption = commandLine.getOption(QEmuMonitorOption.class);
034        if (monitorOption == null)
035            return null;
036        String monitorChardev = monitorOption.chardev;
037        for (QEmuChardevOption chardevOption : commandLine.getOptions(QEmuChardevOption.class)) {
038            if (chardevOption.id.equals(monitorChardev)) {
039                if (chardevOption.device instanceof TcpCharDevice) {
040                    TcpCharDevice tcpDevice = (TcpCharDevice) chardevOption.device;
041                    return tcpDevice.getAddress();
042                }
043            }
044        }
045        return null;
046    }
047}
