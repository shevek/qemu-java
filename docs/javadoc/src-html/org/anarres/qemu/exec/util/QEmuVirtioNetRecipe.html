005package org.anarres.qemu.exec.util;
006
007import javax.annotation.Nonnull;
008import org.anarres.qemu.exec.QEmuDeviceOption;
009import org.anarres.qemu.exec.QEmuNetdevOption;
010
011/**
012 *
013 * @author shevek
014 */
015public class QEmuVirtioNetRecipe extends QEmuOptionsList {
016
017    // tap,fd=27,id=hostnet0,vhost=on,vhostfd=28
018    public final QEmuNetdevOption.Tap netdevOption;
019    // virtio-net-pci,netdev=hostnet0,id=net0,mac=fa:16:3e:13:ff:1f,bus=pci.0,addr=0x3
020    public final QEmuDeviceOption.VirtioNet deviceOption;
021
022    public QEmuVirtioNetRecipe(QEmuIdAllocator allocator) {
023        int id = allocator.newNetworkIndex();
024        netdevOption = new QEmuNetdevOption.Tap();
025        netdevOption
026                .withId("backend-net-" + id);
027        add(netdevOption);
028        deviceOption = new QEmuDeviceOption.VirtioNet();
029        deviceOption
030                .withId("virtio-net-" + id)
031                .withAddress(allocator)
032                .withProperty(QEmuDeviceOption.VirtioNet.PROP_NETDEV, netdevOption.id);
033        add(deviceOption);
034    }
035
036    @Nonnull
037    public QEmuVirtioNetRecipe withMac(@Nonnull String mac) {
038        deviceOption.withMac(mac);
039        return this;
040    }
041
042    @Nonnull
043    public QEmuVirtioNetRecipe withAddress(@Nonnull String address) {
044        deviceOption.withAddress(address);
045        return this;
046    }
047
048    @Nonnull
049    public QEmuVirtioNetRecipe withProperty(@Nonnull String key, @Nonnull String value) {
050        deviceOption.withProperty(key, value);
051        return this;
052    }
053}
