006package org.anarres.qemu.image;
007
008import com.fasterxml.jackson.databind.ObjectMapper;
009import com.google.common.base.Preconditions;
010import com.google.common.io.ByteStreams;
011import java.io.File;
012import java.io.IOException;
013import javax.annotation.Nonnull;
014
015/**
016 *
017 * @author shevek
018 */
019public class QEmuImage {
020
021    private final File file;
022
023    public QEmuImage(@Nonnull File file) {
024        this.file = Preconditions.checkNotNull(file);
025    }
026
027    public QEmuImage(@Nonnull String path) {
028        this(new File(path));
029    }
030
031    @Nonnull
032    public File getFile() {
033        return file;
034    }
035
036    @Nonnull
037    public QEmuImageInfo query() throws IOException {
038        ProcessBuilder builder = new ProcessBuilder("qemu-img", "info", "--output=json", file.getAbsolutePath());
039        Process process = builder.start();
040        byte[] data = ByteStreams.toByteArray(process.getInputStream());
041        ObjectMapper mapper = new ObjectMapper();
042        return mapper.readValue(data, QEmuImageInfo.class);
043    }
044
045    public void create(@Nonnull QEmuImageFormat format, long size) throws IOException {
046        ProcessBuilder builder = new ProcessBuilder("qemu-img", "create", "-f", format.name(), file.getAbsolutePath(), String.valueOf(size));
047        Process process = builder.start();
048        ByteStreams.copy(process.getInputStream(), System.err);
049    }
050
051    /**
052     * Creates this image.
053     *
054     * backingFile is not referenced absolutely here. If you want it referenced
055     * absolutely, canonicalize with {@link File#getAbsoluteFile()}.
056     */
057    public void create(@Nonnull QEmuImageFormat format, @Nonnull File backingFile) throws IOException {
058        ProcessBuilder builder = new ProcessBuilder("qemu-img", "create", "-f", format.name(), "-b", backingFile.getPath(), file.getAbsolutePath());
059        Process process = builder.start();
060        ByteStreams.copy(process.getInputStream(), System.err);
061    }
062
063    public void delete() throws IOException {
064        if (file.exists())
065            if (!file.delete())
066                throw new IOException("Unable to delete " + file);
067    }
068}
